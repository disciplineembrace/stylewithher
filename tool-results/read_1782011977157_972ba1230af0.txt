   388→interface ProductFormState {

   390→  basePrice: string; salePrice: string; material: string; care: string;
   391→  isFeatured: boolean; isTrending: boolean; isNewArrival: boolean; isBestSeller: boolean;

   397→  isFeatured: false, isTrending: false, isNewArrival: false, isBestSeller: false,

   399→
   400→function ProductForm({ form, setForm, categories, loading, onSubmit, submitLabel }: {
   401→  form: ProductFormState; setForm: (f: ProductFormState) => void; categories: any[];
   402→  loading: boolean; onSubmit: () => void; submitLabel: string;
   403→}) {
   404→  const updateField = (field: keyof ProductFormState, value: string | boolean) => {
   405→    setForm({ ...form, [field]: value })
   406→    if (field === 'name') {
   407→      const slug = value.toString().toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/(^-|-$)/g, '')
   408→      setForm((prev: ProductFormState) => ({ ...prev, slug }))
   409→    }
   410→  }
   411→
   412→  return (
   413→    <div className="space-y-4 max-h-[70vh] overflow-y-auto pr-1">
   414→      <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   415→        <div className="space-y-2">
   416→          <Label htmlFor="pf-name">Product Name *</Label>
   417→          <Input id="pf-name" value={form.name} onChange={(e) => updateField('name', e.target.value)} placeholder="Product name" />
   418→        </div>
   419→        <div className="space-y-2">
   420→          <Label htmlFor="pf-slug">Slug *</Label>
   421→          <Input id="pf-slug" value={form.slug} onChange={(e) => updateField('slug', e.target.value)} placeholder="product-slug" />
   422→        </div>
   423→      </div>
   424→      <div className="space-y-2">
   425→        <Label htmlFor="pf-desc">Description</Label>
   426→        <Textarea id="pf-desc" value={form.description} onChange={(e) => updateField('description', e.target.value)} placeholder="Product description..." rows={3} />
   427→      </div>
   428→      <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   429→        <div className="space-y-2">
   430→          <Label>Category *</Label>
   431→          <Select value={form.categoryId} onValueChange={(v) => updateField('categoryId', v)}>
   432→            <SelectTrigger><SelectValue placeholder="Select category" /></SelectTrigger>
   433→            <SelectContent>
   434→              {categories.map((c: any) => (
   435→                <SelectItem key={c.id} value={c.id}>{c.name}</SelectItem>
   436→              ))}
   437→            </SelectContent>
   438→          </Select>
   439→        </div>
   440→        <div className="space-y-2">
   441→          <Label>Gender</Label>
   442→          <Select value={form.gender} onValueChange={(v) => updateField('gender', v)}>
   443→            <SelectTrigger><SelectValue /></SelectTrigger>
   444→            <SelectContent>
   445→              <SelectItem value="women">Women</SelectItem>
   446→              <SelectItem value="men">Men</SelectItem>
   447→              <SelectItem value="unisex">Unisex</SelectItem>
   448→              <SelectItem value="couple">Couple</SelectItem>
   449→            </SelectContent>
   450→          </Select>
   451→        </div>
   452→      </div>
   453→      <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   454→        <div className="space-y-2">
   455→          <Label htmlFor="pf-base">Base Price (₹) *</Label>
   456→          <Input id="pf-base" type="number" value={form.basePrice} onChange={(e) => updateField('basePrice', e.target.value)} placeholder="999" />
   457→        </div>
