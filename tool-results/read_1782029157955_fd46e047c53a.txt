   403→      const res = await fetch('/api/admin', { headers: authHeaders() })
   404→      if (!res.ok) throw new Error('Failed to fetch')
   405→      const json = await res.json()

   408→      setOrders(json.recentOrders || [])
   409→    } catch (err: any) {
   410→      showToast(err.message || 'Error loading orders', 'error')
   411→    } finally {
   412→      setLoading(false)
   413→    }
   414→  }, [showToast])

   417→

   421→      const res = await fetch(`/api/orders/${orderId}`, {
   422→        method: 'PUT',
   423→        headers: { ...authHeaders(), 'Content-Type': 'application/json' },
   424→        body: JSON.stringify({ status: newStatus }),
   425→      })
   426→      if (!res.ok) { const e = await res.json(); throw new Error(e.error || 'Update failed') }
   427→      showToast(`Order status updated to ${newStatus}`)
   428→      fetchOrders()
   429→    } catch (err: any) {
   430→      showToast(err.message || 'Error updating order', 'error')
   431→    } finally {
   432→      setUpdatingId(null)
   433→    }
   434→  }
   435→
   436→  if (loading) {
   437→    return <div className="space-y-4">{[...Array(5)].map((_, i) => <Skeleton key={i} className="h-16 w-full" />)}</div>
   438→  }
   439→
   440→  return (
   441→    <div className="space-y-4">
   442→      <div className="flex items-center justify-between">
   443→        <h2 className="text-xl font-bold">Orders ({orders.length})</h2>

   446→        </Button>
   447→      </div>

   449→      <div className="bg-white rounded-xl border shadow-sm overflow-hidden">
   450→        <div className="max-h-[65vh] overflow-y-auto">
   451→          <Table>
   452→            <TableHeader>

   455→                <TableHead className="text-xs">Customer</TableHead>

   458→                <TableHead className="text-xs">Total</TableHead>
   459→                <TableHead className="text-xs">Status</TableHead>

   463→            </TableHeader>
